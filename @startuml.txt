abstract class Person {
    #ID : string
    #Name : string
    #Email : string
    -Password : string
    +Person(name:string, email:string, password:string)
    +disaplayDashboard () : void
    +getID() : string const
    +getInfo() : void
    +checkPassword () : bool const
}

class Student {
    -StudentNumber : static int
    -enrolledCourses : vector<Course*>
    +Student(ame:string, email:string, password:string){StudentNumber++}
    +getEnrolledCourses() : void
}

class Lecturer {
    -LecturerNumber : static int
    +Lecturer(ame:string, email:string, password:string){LecturerNumber++}
}

class Administrator {
    -AdministratorNumber : static int
    +Administrator(ame:string, email:string, password:string){AdministratorNumber++}
}

Student --up-|> Person
Lecturer -up--|> Person
Administrator --up-|> Person

Person -up[hidden]---- Course

abstract class Course {
    -Code : string
    -Title : string
    -CreditValue : int
    -Capacity : int
    -Lecturer : Lecturer*
    
    ' MODIFIED FOR FR3.4: Changed to vector<Course*> to handle chains
    -Prerequisites : vector<Course*> 
    
    +Course(code: string, title: string, credits: int, cap: int, lec: Lecturer*)
    +DisplayInfo() : void
    
    ' ADDED FOR FR3.4: Function to recursively check the chain
    +hasMetPrerequisites(s: Student*) : bool 
}

class Lecture {
    ' Removed the string prerequisite
    +Lecture(code: string, title: string, credits: int, cap: int, lec: Lecturer*, prereq: vector<Course*>)
}

class Lab {
    ' Removed the string prerequisite
    +Lab(code: string, title: string, credits: int, cap: int, lec: Lecturer*, prereq: vector<Course*>)
}

class Project {
    ' Removed the string prerequisite
    +Project(code: string, title: string, credits: int, cap: int, lec: Lecturer*, prereq: vector<Course*>)
}

Lecture --down-|> Course
Lab -down--|> Course
Project --down-|> Course

class Timetable {
    -timeslots : vector<TimeSlot>
    +addtimeslot(slot: TimeSlot) : void
    +hasClash(other: Timetable) : bool
}

class TimeSlot {
    -day : string
    -startTime : string
    -endTime : string
    -location : string
    +overlapWith(other: TimeSlot) : bool
}

class EnrollmentTemp {
    - studentId : string
    - selectedCourses : vector<Course*>
    + addCourse(course: Course*) : void
    + removeCourse(course: Course*) : void
    + showallCourses() : vector<string>
    + finalizeRegistration() : bool
}

Timetable --* Course
TimeSlot -right-* Timetable
EnrollmentTemp .down.> Student
EnrollmentTemp .up.> Course
